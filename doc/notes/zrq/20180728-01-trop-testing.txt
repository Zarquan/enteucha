# Modify the VM memory allocation from 16G to 64G.
#[user@trop03]

    source "${HOME}/libvirt.settings" 

    virsh \
        -c "${connection:?}" \
        setmaxmem \
            "${vmname:?}" '64GB' \
            --config

    virsh \
        -c "${connection:?}" \
        setmem \
            "${vmname:?}" '64GB' \
            --config

    virsh \
        -c "${connection:?}" \
        setvcpus \
            "${vmname:?}" 8 \
            --maximum \
            --config

    virsh \
        -c "${connection:?}" \
        setvcpus \
            "${vmname:?}" 8 \
            --config

    virsh \
        -c "${connection:?}" \
        shutdown \
            "${vmname:?}"

    sleep 20
    
    virsh \
        -c "${connection:?}" \
        start \
            "${vmname:?}"

# -----------------------------------------------------
# Checkout a copy of our source code.
#[user@builder]

    source "${HOME:?}/enteucha.settings"
    if [ ! -e "${ENTEUCHA_CODE:?}" ]
    then
        if [ ! -e "$(dirname ${ENTEUCHA_CODE:?})" ]
        then
            mkdir -p "$(dirname ${ENTEUCHA_CODE:?})"
        fi
        pushd "$(dirname ${ENTEUCHA_CODE:?})"
            git clone "${ENTEUCHA_REPO:?}" "$(basename ${ENTEUCHA_CODE:?})"

        popd
    else
        pushd "${ENTEUCHA_CODE:?}"
            git pull
        popd
    fi

# -----------------------------------------------------
# Set the git username and email.
#[user@builder]

    source "${HOME:?}/enteucha.settings"
    pushd "${ENTEUCHA_CODE:?}"

        git config user.email "ident@example.com"
        git config user.name  "Example Identity"

    popd
